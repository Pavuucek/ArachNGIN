295  public void DoCommand(string command)
296  {
297  string[] strCmdLine = StringUtils.StringSplit(command, " "); // cely prikaz
298  string cmd = strCmdLine[0];
299  string[] parArray;
300  string parStr;
301  if(strCmdLine.Length == 1)
302  {
303  // pouze 1 slovo = prikaz bez parametru
304  parArray = new string[0];
305  parStr = "";
306  }
307  else
308  {
309  parArray = new string[strCmdLine.Length-1];
310  parStr = "";
311  for (int i = 1; i <= strCmdLine.Length-1 ;i++ )
312  {
313  parArray[i-1] = strCmdLine[i];
314  parStr += strCmdLine[i]+" ";
315  }
316  }
317 
318  // zapiseme do outputu
319  if (_echoCommands) Write("Command: " + command);
320 
321  // vyvolame event
322  if(OnCommandEntered != null)
323  {
324  var ea = new CommandEnteredEventArgs(cmd, parArray, parStr.Trim());
325  OnCommandEntered(this,ea);
326  }
327  }

351  private void InternalCommands(object sender, CommandEnteredEventArgs e)
352  {
353  if(_processInternalCommands)
354  {
355  switch(e.Command.ToLower())
356  {
357  case "cls":
358  _consoleForm.lstLogSeparate.Items.Clear();
359  _consoleForm.lstLogPlain.Items.Clear();
360  break;
361  case "echot":
362  Write(DateTime.Now, e.ParamString);
363  break;
364  case "echo":
365  WriteNoTime(e.ParamString);
366  break;
367  case "savelog":
368  if (string.IsNullOrEmpty(e.ParamString)) break;
369  try
370  {
371  StringCollections.SaveToFile(e.ParamString, _consoleForm.lstLogSeparate.Items);
372  }
373  catch (Exception)
374  {
375  Write("Unable to save to file! " + e.ParamString);
376  }
377  break;
378  }

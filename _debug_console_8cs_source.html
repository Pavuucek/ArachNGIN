311  public void DoCommand(string command)
312  {
313  var strCmdLine = StringUtils.StringSplit(command, " "); // cely prikaz
314  var cmd = strCmdLine[0];
315  string[] parArray;
316  string parStr;
317  if (strCmdLine.Length == 1)
318  {
319  // pouze 1 slovo = prikaz bez parametru
320  parArray = new string[0];
321  parStr = "";
322  }
323  else
324  {
325  parArray = new string[strCmdLine.Length - 1];
326  parStr = "";
327  for (var i = 1; i <= strCmdLine.Length - 1; i++)
328  {
329  parArray[i - 1] = strCmdLine[i];
330  parStr += strCmdLine[i] + " ";
331  }
332  }
333 
334  // zapiseme do outputu
335  if (EchoCommands) Write("Command: " + command);
336 
337  // vyvolame event
338  if (OnCommandEntered != null)
339  {
340  var ea = new CommandEnteredEventArgs(cmd, parArray, parStr.Trim());
341  OnCommandEntered(this, ea);
342  }
343  }

44  internal static StackFrame FindStackFrame()
45  {
46  var stackTrace = new StackTrace();
47  for (var i = 0; i < stackTrace.GetFrames().Count(); i++)
48  {
49  var methodBase = stackTrace.GetFrame(i).GetMethod();
50  var name = MethodBase.GetCurrentMethod().Name;
51  if (!methodBase.Name.Equals("Trace") && !methodBase.Name.Equals(name))
52  return new StackFrame(i, true);
53  }
54  return null;

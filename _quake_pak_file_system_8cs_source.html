66  private void ReadPakFiles()
67  {
68  if (_iPakCount == 0) return;
69  _pakFat = new List<List<string>>();
70  _indexFat = new List<List<string>>();
71  for (var i = 0; i < _iPakCount; i++)
72  {
73  var q = new QuakePakFile(_sDir + _lPakFiles[i], false);
74 
75  _pakFat[i] = q.PakFileList;
76  if (!q.PakFileExists(PakIndexFileName)) continue;
77  using (var st = new MemoryStream())
78  {
79  q.ExtractStream(PakIndexFileName, st);
80  st.Position = 0;
81  var tr = new StreamReader(st);
82  string line;
83  while ((line = tr.ReadLine()) != null) _indexFat[i].Add(line);
84  }

117  public bool AskFile(string sFile)
118  {
119  var sIndexfile = string.Empty;
120  sFile = ReplaceSlashesOut(sFile); // jen pro jistotu
121  if (File.Exists(_sTemp + sFile))
122  return true;
123  // soubor v adresari ma prioritu
124  if (File.Exists(_sDir + sFile))
125  {
126  var sFullpath = _sTemp + sFile;
127  Directory.CreateDirectory(Path.GetDirectoryName(sFullpath));
128  File.Copy(_sDir + sFile, sFullpath, true);
129  return true;
130  }
131 
132  // ted uz pracujeme s pakem, takze lomitka do unixovyho tvaru :-)
133  sFile = ReplaceSlashesIn(sFile);
134  // soubor musime najit v paku
135  if (_pakFat == null || _indexFat == null) return false;
136 
137  foreach (var indexItem in _indexFat)
138  {
139  var sIndexline = "";
140  // prohledame fat index
141  foreach (var sLine in indexItem)
142  if (
143  sLine.ToLower(CultureInfo.InvariantCulture)
144  .Contains(sFile.ToLower(CultureInfo.InvariantCulture) + "="))
145  sIndexline = sLine;
146  // rozdelit radku indexu na fajl jmeno souboru v paku a skutecne jmeno
147  var aIndexline = sIndexline.Split('=');
148  if (aIndexline.Length <= 1) continue;
149  sIndexfile = aIndexline[0];
150  sFile = aIndexline[1];
151  }
152  for (var i = 0; i < _pakFat.Count; i++)
153  {
154  if (!_pakFat[i].Contains(sFile)) continue;
155  var sFullpath = _sTemp;
156  if (sIndexfile != string.Empty) sFullpath += sIndexfile;
157  else sFullpath += sFile;
158  // prevest lomitka :-)
159  sFullpath = ReplaceSlashesOut(sFullpath);
160  Directory.CreateDirectory(Path.GetDirectoryName(sFullpath));
161  var q = new QuakePakFile(_sDir + _lPakFiles[i], false);
162  q.ExtractFile(sFile, sFullpath);
163  if (File.Exists(sFullpath)) return true;
164  }
165 
166  return false;

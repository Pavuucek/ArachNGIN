82  private void ReadPakFiles()
83  {
84  if (_iPakCount == 0) return;
85  _pakFat = new StringCollection[_iPakCount];
86  _indexFat = new StringCollection[_iPakCount];
87  for (int i = 0; i < _iPakCount; i++)
88  {
89  var q = new QuakePakFile(_sDir + _lPakFiles[i], false);
90  _pakFat[i] = q.PakFileList;
91  _indexFat[i] = new StringCollection();
92  if (q.PakFileExists(PakIndexFileName))
93  {
94  Stream st = new MemoryStream();
95  q.ExtractStream(PakIndexFileName, st);
96  st.Position = 0;
97  var tr = new StreamReader(st);
98  string line;
99  while ((line = tr.ReadLine()) != null)
100  {
101  _indexFat[i].Add(line);
102  }
103  st.Close();
104  }
105  }

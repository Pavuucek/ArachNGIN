166  private void SetValue(SettingsPropertyValue propertyValue)
167  {
168  XmlNode targetNode = IsGlobal(propertyValue.Property)
169  ? GlobalSettingsNode
170  : LocalSettingsNode;
171  XmlNode settingNode = targetNode.SelectSingleNode(string.Format("setting[@name='{0}']", propertyValue.Name));
172  if (settingNode != null)
173  settingNode.InnerText = propertyValue.SerializedValue.ToString();
174  else
175  {
176  settingNode = RootDocument.CreateElement("setting");
177  XmlAttribute nameAttribute = RootDocument.CreateAttribute("name");
178  nameAttribute.Value = propertyValue.Name;
179  if (settingNode.Attributes != null) settingNode.Attributes.Append(nameAttribute);
180  settingNode.InnerText = propertyValue.SerializedValue.ToString();
181  targetNode.AppendChild(settingNode);
182  }
183  }
184 
185  private string GetValue(SettingsProperty property)
186  {
187  XmlNode targetNode = IsGlobal(property) ? GlobalSettingsNode : LocalSettingsNode;
188  XmlNode settingNode = targetNode.SelectSingleNode(string.Format("setting[@name='{0}']", property.Name));
189  if (settingNode == null)
190  return property.DefaultValue != null ? property.DefaultValue.ToString() : string.Empty;
191  return settingNode.InnerText;
192  }

194  private bool IsGlobal(SettingsProperty property)
195  {
196  foreach (DictionaryEntry attribute in property.Attributes)
197  {
198  if ((Attribute) attribute.Value is SettingsManageabilityAttribute)
199  return true;
200  }
201  return false;
202  }

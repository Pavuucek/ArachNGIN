181  private void SetValue(SettingsPropertyValue propertyValue)
182  {
183  var targetNode = IsGlobal(propertyValue.Property)
184  ? GlobalSettingsNode
185  : LocalSettingsNode;
186  var settingNode = targetNode.SelectSingleNode(string.Format("setting[@name='{0}']", propertyValue.Name));
187  if (settingNode != null)
188  {
189  settingNode.InnerText = propertyValue.SerializedValue.ToString();
190  }
191  else
192  {
193  settingNode = RootDocument.CreateElement("setting");
194  var nameAttribute = RootDocument.CreateAttribute("name");
195  nameAttribute.Value = propertyValue.Name;
196  if (settingNode.Attributes != null) settingNode.Attributes.Append(nameAttribute);
197  settingNode.InnerText = propertyValue.SerializedValue.ToString();
198  targetNode.AppendChild(settingNode);
199  }
200  }
201 
207  private string GetValue(SettingsProperty property)
208  {
209  var targetNode = IsGlobal(property) ? GlobalSettingsNode : LocalSettingsNode;
210  var settingNode = targetNode.SelectSingleNode(string.Format("setting[@name='{0}']", property.Name));
211  if (settingNode == null)
212  return property.DefaultValue != null ? property.DefaultValue.ToString() : string.Empty;
213  return settingNode.InnerText;
214  }

216  private static bool IsGlobal(SettingsProperty property)
217  {
218  foreach (DictionaryEntry attribute in property.Attributes)
219  if ((Attribute)attribute.Value is SettingsManageabilityAttribute)
220  return true;
221  return false;
222  }

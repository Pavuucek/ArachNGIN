178  private void FormatXml(XmlReader reader, string fileName)
179  {
180  while (reader.Read())
181  {
182  if (reader.NodeType == XmlNodeType.Element)
183  {
184  string sName = "";
185  string sValue = "";
186  if (reader.Name == "Item")
187  {
188  if (reader.HasAttributes)
189  {
190  while (reader.MoveToNextAttribute())
191  {
192  if (reader.Name == "Name")
193  {
194  sName = reader.Value;
195  }
196  else if (reader.Name == "Value")
197  {
198  sValue = reader.Value;
199  }
200  }
201  }
202  if (sName != "") _mSettingstable.Add(sName, sValue);
203  }
204  }
205  }
206  }

90  private static void Main(string[] args)
91  {

95  DisplayHelp();

100  if (!File.Exists(args[1])) QuakePakFile.CreateNewPak(args[1]);
101  var pak = new QuakePakFile(args[1], true);
102 
103  switch (args[0])
104  {
105  case "a":
106  if (!AddSingleFile(args, pak)) return;
107  break;
108 
109  case "d":
110  if (!AddDirectory(args, pak)) return;
111  break;
112 
113  case "e":
114  if (!ExtractSingleFile(args, pak)) return;
115  break;
116 
117  case "x":
118  if (!ExtractAllFiles(args, pak)) return;
119  break;
120  // pridat a indexovat
121  case "ia":
122  if (args.Length < 3)
123  {
124  NotEnoughArgs();
125  return;
126  }
127  LoadIndex(pak);
128 
129  if (ExistsInIndex(args[2]))
130  {
131  var g = Guid.NewGuid();
132  ReplaceInIndex(args[2], g.ToString());
133  pak.AddFile(args[2], g.ToString());
134  }
135  else
136  {
137  var s = AddIndexedFile(args[2]);
138  // TODO: tohle neudělá nic. AddFile neumí nahrazovat :-)
139  pak.AddFile(args[2], s);
140  }
141  //
142  FinishIndex();
143  using (Stream ms = new MemoryStream())
144  {
145  PakIndex.SaveToStream(ms);
146  ms.Seek(0, SeekOrigin.Begin);
147  pak.AddStream(ms, "(pak-index)", true);

150  }

326  private void LoadIniToDataSet()
327  {
328  DataStorage = new DataSet();
329  var file = new FileInfo(_mFileName);
330  DataStorage.DataSetName = file.Name.Remove(file.Name.IndexOf(file.Extension, StringComparison.Ordinal),
331  file.Extension.Length);
332  if (!file.Exists) return;
333  DataTable table = null;
334  DataRow row = null;
335  var addRow = false;
336  var skipSection = false;
337  using (var fileStream = new StreamReader(_mFileName))
338  {
339  var readLine = fileStream.ReadLine();
340  while (readLine != null)
341  {
342  readLine = readLine.Trim();
343  if (readLine != "" && !readLine.StartsWith(";"))
344  // section marker
345  if (readLine.StartsWith("[") && readLine.EndsWith("]"))
346  {
347  // table.rows.add moved from here
348  if (addRow) table.Rows.Add(row);
349  readLine = readLine.TrimStart('[');
350  readLine = readLine.TrimEnd(']');
351  skipSection = true;
352  table = DataStorage.Tables[readLine];
353  if (table == null)
354  {
355  table = new DataTable(readLine);
356  DataStorage.Tables.Add(table);
357  row = table.NewRow();
358  skipSection = false;
359  }
360  addRow = false;
361  }
362  else // normal line
363  {
364  if (!skipSection)
365  {
366  var splitLine = readLine.Split('=');
367  if (table != null && table.Columns[splitLine[0]] == null)
368  {
369  table.Columns.Add(splitLine[0]);
370  if (splitLine.Length == 2)
371  row[splitLine[0]] = splitLine[1];
372  else
373  row[splitLine[0]] = "";
374  addRow = true;
375  }
376  }
377  }
378  //... to here
379  readLine = fileStream.ReadLine();
380  }
381  // ... and here deleted
382  if (addRow) table.Rows.Add(row);
383  }
384  }

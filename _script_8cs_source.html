89  statement);

120  stackScriptBlocks.Push(scriptBlock);

153  break;

155  if (stackWhileBlocks.Count == 0)
156  throw new ScriptException("ENDWHILE without matching WHILE", statement);
157 
158  // pop while block
159  stackScriptBlocks.Pop();
160 
161  // set next statement to next in current block
162  whileBlock = stackWhileBlocks.Pop();

164

166 
167  break;
168  default:

172  stackScriptBlocks.Push(scriptBlock);

174  }

235 
236  // parse to statements
237  foreach (String strStatement in listStatements)
238  _mListStatements.Add(new Statement(this, 0, strStatement));
239 
240  // includes dictionary
241  Dictionary<String, bool> dictIncludes = new Dictionary<string, bool>();
242 
243  // process statements for INCLUDEs
244  for (int iIndex = 0; iIndex < _mListStatements.Count; iIndex++)
245  {
246  // skip anything that is not a control statement
247  Statement statement = _mListStatements[iIndex];
248  if (statement.Type != StatementType.Control) continue;
249 
250  // skip statements that are not INCLUDE
251  Token token = statement.Tokens[0];
252  if (token.Type != TokenType.Include) continue;
253 
254  // remove INCLUDE statement
255  _mListStatements.RemoveAt(iIndex);
256 
257  // determine included script name
258  String strIncludeName = statement.Tokens[1].Value.ToString();
259 
260  // ignore if already included
261  if (dictIncludes.ContainsKey(strIncludeName.ToUpper())) continue;
262 
263  List<Statement> listStatementsIncluded = new List<Statement>();
264  foreach (String strStatement in _mScriptManager.Loader.LoadScript(strIncludeName))
265  listStatementsIncluded.Add(new Statement(this, 0, strStatement));
266 
267  _mListStatements.InsertRange(iIndex, listStatementsIncluded);

269  // mark include file as included
270  dictIncludes[strIncludeName.ToUpper()] = true;
271  }

275  _mListStatements[iIndex].Line = iIndex + 1;

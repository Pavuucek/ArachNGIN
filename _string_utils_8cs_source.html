46  public static string[] StringSplit(string wholeString, string delimiter)
47  {
48  var r = new Regex("(" + delimiter + ")");
49  string[] s = r.Split(wholeString);
50  int iHalf = Convert.ToInt16((s.GetUpperBound(0)/2) + 1);
51  var res = new string[iHalf];
52  int j = 0;
53  for (int i = 0; i <= s.GetUpperBound(0); i++)
54  {
55  if (s[i] != delimiter)
56  {
57  res[j] = s[i];
58  j++;
59  }
60  }
61  return res;
62  }

191  private static TreeNode LastNode(TreeView treeView, char pathSeparator, string path, TreeNode lastNode)
192  {
193  string subPathAgg = string.Empty;
194  foreach (string subPath in path.Split(pathSeparator))
195  {
196  subPathAgg += subPath + pathSeparator;
197  TreeNode[] nodes = treeView.Nodes.Find(subPathAgg, true);
198  if (nodes.Length == 0)
199  if (lastNode == null)
200  lastNode = treeView.Nodes.Add(subPathAgg, subPath);
201  else
202  lastNode = lastNode.Nodes.Add(subPathAgg, subPath);
203  else
204  lastNode = nodes[0];
205  }
206  return lastNode;
207  }

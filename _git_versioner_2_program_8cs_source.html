82  private static string FindGitBinary()
83  {
84  string git = null;
85  RegistryKey key;
86 
87  // Try the PATH environment variable
88 
89  string pathEnv = Environment.GetEnvironmentVariable("PATH");
90  if (pathEnv != null)
91  foreach (string dir in pathEnv.Split(Path.PathSeparator))
92  {
93  string sdir = dir;
94  if (sdir.StartsWith("\"") && sdir.EndsWith("\""))
95  {
96  // Strip quotes (no Path.PathSeparator supported in quoted directories though)
97  sdir = sdir.Substring(1, sdir.Length - 2);
98  }
99  git = Path.Combine(sdir, GitExeName);
100  if (File.Exists(git)) break;
101  }
102  if (!File.Exists(git)) git = null;
103 
104 
105  // Read registry uninstaller key
106  if (git == null)
107  {
108  key = Registry.LocalMachine.OpenSubKey(@"SOFTWARE\Microsoft\Windows\CurrentVersion\Uninstall\Git_is1");
109  if (key != null)
110  {
111  object loc = key.GetValue("InstallLocation");
112  if (loc is string)
113  {
114  git = Path.Combine((string) loc, Path.Combine("bin", GitExeName));
115  if (!File.Exists(git)) git = null;
116  }
117  }
118  }
119 
120  // Try 64-bit registry key
121  if (git == null && Is64Bit)
122  {
123  key =
124  Registry.LocalMachine.OpenSubKey(
125  @"SOFTWARE\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall\Git_is1");
126  if (key != null)
127  {
128  object loc = key.GetValue("InstallLocation");
129  if (loc is string)
130  {
131  git = Path.Combine((string) loc, Path.Combine("bin", GitExeName));
132  if (!File.Exists(git)) git = null;
133  }
134  }
135  }
136 
137  // Search program files directory
138  if (git == null)
139  {
140  foreach (
141  string dir in
142  Directory.GetDirectories(Environment.GetFolderPath(Environment.SpecialFolder.ProgramFiles),
143  "git*"))
144  {
145  git = Path.Combine(dir, Path.Combine("bin", GitExeName));
146  if (!File.Exists(git)) git = null;
147  }
148  }
149 
150  // Try 32-bit program files directory
151  if (git != null || !Is64Bit) return git;
152  foreach (string dir in Directory.GetDirectories(ProgramFilesX86(), "git*"))
153  {
154  git = Path.Combine(dir, Path.Combine("bin", GitExeName));
155  if (!File.Exists(git)) git = null;
156  }
157 
158  return git;
159  }

167  private static void WriteInfo(string sFile, bool backup = true)
168  {
169  if (!File.Exists(sFile))
170  {
171  Console.WriteLine("Unable to find file {0}", sFile);
172  return;
173  }
174  string bkp = sFile + ".gwbackup";
175  //if (File.Exists(bkp)) return false;
176  try
177  {
178  // zapis
179  if (backup) File.Copy(sFile, bkp, true);
180  GitResult git = GetVersionInfo(Path.GetDirectoryName(Path.GetFullPath(sFile)));
181  using (var infile = new StreamReader(bkp, Encoding.Default, true))
182  {
183  Console.WriteLine("Reading {0}...", sFile);
184  bool append = true;
185  using (var outfile = new StreamWriter(sFile, false, infile.CurrentEncoding))
186  {
187  Console.WriteLine("Writing {0}", sFile);
188  while (!infile.EndOfStream)
189  {
190  string l = infile.ReadLine();
191  if (l != null && l.Contains("AssemblyInformationalVersion")) append = false;
192  outfile.WriteLine(DoReplace(l, git));

195  if (append)

198  outfile.WriteLine(
199  DoReplace(
200  "[assembly: AssemblyInformationalVersion(\"$Branch$:$MajorVersion$.$MinorVersion$.$Revision$-$Commit$-$ShortHash$\")]",
201  git));
202  }
203  }
204  }
205  }
206  catch (Exception e)
207  {
208  Console.WriteLine("Error: '{0}' in '{1}'", e.Message, e.Source);
209  }

218  private static GitResult GetVersionInfo(string workDir)
219  {
220  Console.WriteLine("Getting version info for {0}", workDir);
221  string lines = ExecGit(workDir, "describe --long --tags --always");
222  GitResult r;
223  r.MajorVersion = 0;
224  r.MinorVersion = 0;
225  r.Revision = 0;
226  r.Commit = 0;
227  r.ShortHash = "";
228  // ocekavany retezec ve formatu: 1.7.6-235-g0a52e4b
229  //lines = "g0a52e4b";
230  string[] part1 = lines.Split('-');
231  if (part1.Length >= 3)
232  {
233  // druhou cast rozdelit po teckach
234  string[] part2 = part1[0].Split('.');
235  if (part2.Length > 1)
236  {
237  // delsi nez 1: mame major a minor verzi
238  if (part2.Length > 2)
239  {
240  // mame i revizi
241  try
242  {
243  r.Revision = Convert.ToInt32(part2[2]);
244  }
245  catch
246  {
247  r.Revision = 0;
248  }
249  }
250  try
251  {
252  r.MinorVersion = Convert.ToInt32(part2[1]);
253  }
254  catch
255  {
256  r.MinorVersion = 0;
257  }
258  }
259  // mame jen major verzi
260  try
261  {
262  string s = part2[0].ToLower();
263  // kdyby nahodou nekdo chtel pojmenovavat git tagy v1.0.0 atd (tj zacinajci ne cislem ale v)
264  if (s[0] == 'v')
265  {
266  s = s.Remove(0, 1);
267  }
268  r.MajorVersion = Convert.ToInt32(s);
269  }
270  catch
271  {
272  r.MajorVersion = 0;
273  }
274  }
275  try
276  {
277  r.Commit = Convert.ToInt32(part1[1]);
278  }
279  catch
280  {
281  r.Commit = 0;
282  }
283  try
284  {
285  r.ShortHash = part1[2];
286  }
287  catch
288  {
289  r.ShortHash = lines;
290  }
291  r.ShortHash = r.ShortHash.Trim();
292  //
293  r.Branch = ExecGit(workDir, "rev-parse --abbrev-ref HEAD").Trim();
294  r.LongHash = ExecGit(workDir, "rev-parse HEAD").Trim();
295  Console.WriteLine("Version info: {0}", GitResultToString(r));
296  if (string.IsNullOrEmpty(lines))
297  {
298  Console.WriteLine("Possible error, git output follows:\n {0}", lines);
299  }
300  return r;
301  }

326  private static string ExecGit(string workDir, string parameters)
327  {
328  var psi = new ProcessStartInfo(FindGitBinary(), parameters)
329  {
330  WorkingDirectory = workDir,
331  RedirectStandardOutput = true,
332  UseShellExecute = false
333  };
334  Process p = Process.Start(psi);
335  string r = string.Empty;
336  while (p != null && !p.StandardOutput.EndOfStream)
337  {
338  r += p.StandardOutput.ReadLine() + "\n";
339  }
340  if (p != null && !p.WaitForExit(1000))
341  {
342  p.Kill();
343  }
344  return r;

354  private static string DoReplace(string inString, GitResult gr)
355  {
356  string r = inString.Replace("$MajorVersion$", gr.MajorVersion.ToString(CultureInfo.InvariantCulture));
357  r = r.Replace("$MinorVersion$", gr.MinorVersion.ToString(CultureInfo.InvariantCulture));
358  r = r.Replace("$Revision$", gr.Revision.ToString(CultureInfo.InvariantCulture));
359  r = r.Replace("$Commit$", gr.Commit.ToString(CultureInfo.InvariantCulture));
360  r = r.Replace("$ShortHash$", gr.ShortHash);
361  r = r.Replace("$LongHash$", gr.LongHash);
362  r = r.Replace("$Branch$", gr.Branch);
363  return r;

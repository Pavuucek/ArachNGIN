100  private bool ReadHeader()
101  {
102  _pakStream.Position = 0;
103  string pHeader = StreamHandling.PCharToString(_pakReader.ReadChars(PakId.Length));
104  if (pHeader == StreamHandling.PCharToString(PakId))
105  {
106  // hned za hlavickou je pozice zacatku fatky
107  _pFatstart = _pakReader.ReadInt32();
108  // a pak je pocet souboru * 64
109  _pFilecount = _pakReader.ReadInt32()/64;
110  //
111  // presuneme se na pozici fatky a nacteme ji
112  _pakStream.Position = _pFatstart;
113  _pakFat = new PakFat[_pFilecount];
114  // vymazneme filelist
115  PakFileList.Clear();
116  for (int i = 0; i < _pFilecount; i++)
117  {
118  // my radi lowercase. v tom se lip hleda ;-)
119  string sfile = StreamHandling.PCharToString(_pakReader.ReadChars(56)).ToLower();
120  sfile = sfile.Replace("/", "\\"); // unixovy lomitka my neradi.
121  // pridame soubor do filelistu a do PakFATky
122  PakFileList.Add(sfile);
123  _pakFat[i].FileName = sfile;
124  _pakFat[i].FileStart = _pakReader.ReadInt32();
125  _pakFat[i].FileLength = _pakReader.ReadInt32();
126  }
127  _pakStream.Position = 0;
128  //
129  return true;
130  }
131  _pakStream.Position = 0;
132  return false;
133  }

252  public bool AddStream(Stream stream, string pakFileName, bool writeFat /*=true*/)
253  {
254  // soubor uz existuje --> dal se nebavime!
255  if (PakFileExists(pakFileName)) return false;
256  // novy soubor zapisujeme na pozici fatky
257  _pakStream.Seek(_pFatstart, SeekOrigin.Begin);
258  // vytvorit novou fatku a zapsat do ni novy soubor
259  PakFat[] oldPakFat = _pakFat;
260  _pFilecount = oldPakFat.Length + 1;
261  _pakFat = new PakFat[_pFilecount];
262  oldPakFat.CopyTo(_pakFat, 0);
263  _pakFat[_pFilecount - 1].FileName = pakFileName;
264  _pakFat[_pFilecount - 1].FileLength = (int) stream.Length;
265  _pakFat[_pFilecount - 1].FileStart = _pFatstart;
266  // zapsat soubor
267  StreamHandling.StreamCopy(stream, _pakStream, 0, _pakStream.Position);
268  //StreamHandling.StreamCopyAsync(stream, PakStream);
269  // po dokonceni zapisovani zjistit pozici streamu
270  _pFatstart = (int) _pakStream.Position;
271  //bw.Close();
272  // zapsat fatku
273  if (writeFat)
274  {
275  _pakStream.Seek(PakId.Length, SeekOrigin.Begin);
276  var bw = new BinaryWriter(_pakStream);
277  // zapsat startovni pozici fatky
278  bw.Write(_pFatstart);
279  WriteFat();
280  }
281  return true;
282  }

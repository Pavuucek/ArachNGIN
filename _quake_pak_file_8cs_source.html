117  private bool ReadHeader()
118  {
119  _pakStream.Position = 0;
120  string pHeader = StreamHandling.PCharToString(_pakReader.ReadChars(PakId.Length));
121  if (pHeader == StreamHandling.PCharToString(PakId))
122  {
123  // hned za hlavickou je pozice zacatku fatky
124  _pFatstart = _pakReader.ReadInt32();
125  // a pak je pocet souboru * 64
126  _pFilecount = _pakReader.ReadInt32()/64;
127  //
128  // presuneme se na pozici fatky a nacteme ji
129  _pakStream.Position = _pFatstart;
130  _pakFat = new PakFat[_pFilecount];
131  // vymazneme filelist
132  PakFileList.Clear();
133  for (int i = 0; i < _pFilecount; i++)
134  {
135  // my radi lowercase. v tom se lip hleda ;-)
136  string sfile = StreamHandling.PCharToString(_pakReader.ReadChars(56)).ToLower();
137  sfile = sfile.Replace("/", "\\"); // unixovy lomitka my neradi.
138  // pridame soubor do filelistu a do PakFATky
139  PakFileList.Add(sfile);
140  _pakFat[i].FileName = sfile;
141  _pakFat[i].FileStart = _pakReader.ReadInt32();
142  _pakFat[i].FileLength = _pakReader.ReadInt32();
143  }
144  _pakStream.Position = 0;
145  //
146  return true;
147  }
148  _pakStream.Position = 0;
149  return false;

282  public bool AddStream(Stream stream, string pakFileName, bool writeFat /*=true*/)
283  {
284  // soubor uz existuje --> dal se nebavime!
285  if (PakFileExists(pakFileName)) return false;

289  PakFat[] oldPakFat = _pakFat;
290  _pFilecount = oldPakFat.Length + 1;
291  _pakFat = new PakFat[_pFilecount];
292  oldPakFat.CopyTo(_pakFat, 0);
293  _pakFat[_pFilecount - 1].FileName = pakFileName;
294  _pakFat[_pFilecount - 1].FileLength = (int) stream.Length;
295  _pakFat[_pFilecount - 1].FileStart = _pFatstart;
296  // zapsat soubor

298  //StreamHandling.StreamCopyAsync(stream, PakStream);
299  // po dokonceni zapisovani zjistit pozici streamu
300  _pFatstart = (int) _pakStream.Position;

302  // zapsat fatku

305  _pakStream.Seek(PakId.Length, SeekOrigin.Begin);

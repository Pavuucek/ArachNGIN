115  private bool ReadHeader(Stream pakStream, BinaryReader pakReader)
116  {
117  pakStream.Position = 0;
118  var pHeader = StreamHandling.PCharToString(pakReader.ReadChars(PakId.Length));
119  if (pHeader == StreamHandling.PCharToString(PakId))
120  {
121  // hned za hlavickou je pozice zacatku fatky
122  _pFatstart = pakReader.ReadInt32();
123  // a pak je pocet souboru * 64
124  _pFilecount = pakReader.ReadInt32() / 64;
125  //
126  // presuneme se na pozici fatky a nacteme ji
127  pakStream.Position = _pFatstart;
128  _pakFat = new PakFat[_pFilecount];
129  // vymazneme filelist
130  PakFileList.Clear();
131  for (var i = 0; i < _pFilecount; i++)
132  {
133  // my radi lowercase. v tom se lip hleda ;-)
134  var sfile =
135  StreamHandling.PCharToString(pakReader.ReadChars(56)).ToLower(CultureInfo.InvariantCulture);
136  sfile = sfile.Replace("/", "\\"); // unixovy lomitka my neradi.
137  // pridame soubor do filelistu a do PakFATky
138  PakFileList.Add(sfile);
139  _pakFat[i].FileName = sfile;
140  _pakFat[i].FileStart = pakReader.ReadInt32();
141  _pakFat[i].FileLength = pakReader.ReadInt32();
142  }
143  pakStream.Position = 0;
144  //
145  return true;
146  }
147  pakStream.Position = 0;
148  return false;
149  }

262  private void WriteFat(Stream pakStream)
263  {
264  // naseekovat startovní pozici fatky
265  pakStream.Seek(_pFatstart, SeekOrigin.Begin);
266  using (var bw = new BinaryWriter(pakStream, Encoding.ASCII))
267  {
268  foreach (var item in _pakFat)
269  {
270  // nazev souboru
271  bw.Write(PrepFileNameWrite(item.FileName));
272  bw.Write(item.FileStart);
273  bw.Write(item.FileLength);
274  }
275  // naseekovat na pocet souboru a zapsat
276  pakStream.Seek(PakId.Length + sizeof(int), SeekOrigin.Begin);
277  bw.Write(_pFilecount * 64); // krat 64. z neznamych duvodu
278  }

288  public bool AddStream(Stream stream, string pakFileName, bool writeFat /*=true*/)
289  {
290  // soubor uz existuje --> dal se nebavime!
291  if (PakFileExists(pakFileName)) return false;
292  // mame zakazany zapis
293  if (!WriteAccess) return false;
294  // novy soubor zapisujeme na pozici fatky
295  using (var pakStream = new FileStream(FileName, FileMode.Open, FileAccess.ReadWrite, FileShare.ReadWrite))
296  {
297  pakStream.Seek(_pFatstart, SeekOrigin.Begin);
298  // vytvorit novou fatku a zapsat do ni novy soubor
299  var oldPakFat = _pakFat;
300  _pFilecount = oldPakFat.Length + 1;
301  _pakFat = new PakFat[_pFilecount];
302  oldPakFat.CopyTo(_pakFat, 0);
303  _pakFat[_pFilecount - 1].FileName = pakFileName;
304  _pakFat[_pFilecount - 1].FileLength = (int)stream.Length;
305  _pakFat[_pFilecount - 1].FileStart = _pFatstart;
306  // zapsat soubor
307  StreamHandling.StreamCopy(stream, pakStream, 0, pakStream.Position);
308  // po dokonceni zapisovani zjistit pozici streamu
309  _pFatstart = (int)pakStream.Position;
310  // zapsat fatku
311  if (writeFat)
312  {
313  pakStream.Seek(PakId.Length, SeekOrigin.Begin);
314  using (var bw = new BinaryWriter(pakStream))
315  {
316  // zapsat startovni pozici fatky
317  bw.Write(_pFatstart);
318  WriteFat(pakStream);
319  }
320  }
321  }
322  return true;
323  }

36 // All functions below are actual functions that are part of this
37 // library and do not need any additional dll files. It does not
38 // even require Storm to be able to decompress or compress files.
39 
40 // This library emulates the interface of Lmpqapi and Storm MPQ
41 // functions, so it may be used as a replacement for them in
42 // MPQ extractors/archivers without even needing to recompile
43 // the program that uses Lmpqapi or Storm. It has a few features
44 // not included in Lmpqapi and Storm, such as extra flags for some
45 // functions, setting the locale ID of existing files, and adding
46 // files without having to write them somewhere else first. Also,
47 // MPQ handles used by functions prefixed with "SFile" and "Mpq"
48 // can be used interchangably; all functions use the same type
49 // of MPQ handles. You cannot, however, use handles from this
50 // library with storm or lmpqapi or vice-versa. Doing so will
51 // most likely result in a crash.
52

499  _with1.Revision = 7;

503  if (DllVersion.Major > exeVersion.Major)

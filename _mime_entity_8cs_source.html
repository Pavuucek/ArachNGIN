307  private MailMessageEx ToMailMessageEx(MimeEntity entity)
308  {
309  if (entity == null)
310  {
311  throw new ArgumentNullException("entity");
312  }
313 
314  //parse standard headers and create base email.
315  MailMessageEx message = MailMessageEx.CreateMailMessageFromEntity(entity);
316 
317  if (!string.IsNullOrEmpty(entity.ContentType.Boundary))
318  {
319  message = MailMessageEx.CreateMailMessageFromEntity(entity);
320  BuildMultiPartMessage(entity, message);
321  }//parse multipart message into sub parts.
322  else if (string.Equals(entity.ContentType.MediaType, MediaTypes.MessageRfc822, StringComparison.InvariantCultureIgnoreCase))
323  {
324  //use the first child to create the multipart message.
325  if (entity.Children.Count < 0)
326  {
327  throw new Pop3Exception("Invalid child count on message/rfc822 entity.");
328  }
329 
330  //create the mail message from the first child because it will
331  //contain all of the mail headers. The entity in this state
332  //only contains simple content type headers indicating, disposition, type and description.
333  //This means we can't create the mail message from this type as there is no
334  //internet mail headers attached to this entity.
335  message = MailMessageEx.CreateMailMessageFromEntity(entity.Children[0]);
336  BuildMultiPartMessage(entity, message);
337  } //parse nested message.
338  else
